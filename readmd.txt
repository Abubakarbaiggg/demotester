	class product{
	 public function orderItems(){
	   return $this->hasMany(order_items::class);
	 }
	}
	
	Product::with('orderItems' => function($query){
		$query->where('product_id','!=',null);
	})->get();
	
	Product::doesntHave('orderItems')->get();
